## b.any()

## b.string()

Validates that the input is a string with optional length and pattern constraints.

```typescript
b.string(options?: StringOptions)
```

### Parameters

<ParamField path="options" type="StringOptions" optional>
  Configuration options for the string validator.
  
  <Expandable title="StringOptions">
    <ParamField path="min" type="number" optional>
      Minimum string length (inclusive).
    </ParamField>
    
    <ParamField path="max" type="number" optional>
      Maximum string length (inclusive).
    </ParamField>
    
    <ParamField path="match" type="RegExp" optional>
      Regular expression pattern that the string must match.
    </ParamField>
    
    <ParamField path="notExpectedType" type="string" optional>
      Custom error message when input is not a string. Default: `"Not a string"`
    </ParamField>
    
    <ParamField path="lessThanExpected" type="string" optional>
      Custom error message when string is shorter than `min`. Default: `"String is less than expected"`
    </ParamField>
    
    <ParamField path="gratherThanExpected" type="string" optional>
      Custom error message when string is longer than `max`. Default: `"String is grather than expected"`
    </ParamField>
    
    <ParamField path="notMatchPattern" type="string" optional>
      Custom error message when string doesn't match pattern. Default: `"String does not match pattern"`
    </ParamField>
  </Expandable>
</ParamField>

### Implementation

```typescript validators.ts:61-87
string: (options?: StringOptions) => {
  let min = options?.min ?? undefined;
  let max = options?.max ?? undefined;
  let match = options?.match ?? undefined;

  return {
    validate(input: any) {
      const valid = handle();

      check(valid, typeof input === "string", options?.notExpectedType ?? "Not a string");

      if (min != undefined) {
        check(valid, String(input).length >= min, options?.lessThanExpected ?? "String is less than expected");
      }

      if (max != undefined) {
        check(valid, String(input).length <= max, options?.gratherThanExpected ?? "String is grather than expected");
      }

      if (match != undefined) {
        check(valid, match.test(input), options?.notMatchPattern ?? "String does not match pattern");
      }

      return valid;
    },
  };
}
```

### Usage

<CodeGroup>
```typescript Basic
import { b } from "balid";

const schema = b.string();

schema.validate("Hello World"); // { valid: true, errors: [] }
```

```typescript Min/Max Length
import { b } from "balid";

const minMaxSchema = b.string({ min: 2, max: 10 });

minMaxSchema.validate("a"); // { valid: false, errors: ["String is less than expected"] }
minMaxSchema.validate("hello"); // { valid: true, errors: [] }
```

```typescript Pattern Matching
import { b } from "balid";

const matchSchema = b.string({ match: /hello/ });

matchSchema.validate("hello"); // { valid: true, errors: [] }
matchSchema.validate("world"); // { valid: false, errors: ["String does not match pattern"] }
```
</CodeGroup>
